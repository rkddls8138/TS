2. ## [광명시]가맹점포탈 가입현황 DATA추출

SELECT 
    UI.ID AS ID, 
    BM.SIMPLE_NM AS 상호, 
    XX1.DEC_VARCHAR2_SEL(UI.REAL_NAME, 10, 'CUSTOMER_INFO')  AS 가입자명, 
    BI.BIZ_NUMBER AS 사업자번호,
    UI.REG_DT AS 등록일시,
    BM.ADDR1 || ' ' || BM.ADDR2 AS 주소,
    XX1.DEC_VARCHAR2_SEL(UI.MOBILE_NUMBER, 10, 'CUSTOMER_INFO')  AS 전화번호
FROM KMP.USER_INFO UI
LEFT JOIN KMP.BIZ_MC_MAPPING BMM
    ON BMM.USER_NO = UI.NO
LEFT JOIN KODASP.BC_MERCHANT BM
    ON BM.MCT_MNG_NO = BMM.MCT_ID
INNER JOIN KMP.BIZ_INFO BI
    ON BI.USER_NO = UI.NO
    AND BMM.BIZ_NUMBER = BI.BIZ_NUMBER
WHERE BM.SI_DO_NM = '경기도'
AND BM.SI_GUN_GU_NM LIKE '광명%'
AND TO_CHAR(UI.REG_DT, 'YYYYMMDD') BETWEEN '20201101' AND '20201130'

;

7. ## 인천e음 앱내 혜택플러스 가맹점 지역별 검색결과 가맹점정보

SELECT * FROM (

SELECT

AP.PLACE_NO,

AP.PLAYER_ID,

AP.PLAYER_NO,

TO_CHAR( AP.PLAYER_NO ) AS 가맹점ID,

'인천광역시' AS 인천광역시,

MI.SGG_NM AS 지역구,

BCM.BIZ_LICENSE_NO AS 사업자번호,

MI.VIEWING_PLAYER_NM AS APP노출명,

DSC.DISPLAY_DISCOUNT_NAME AS 할인율,

(MI.PLACE_ADRS1 || ' ' || MI.PLACE_ADRS2) AS 주소,

ag.agent_name AS 사업주명,

(MI.TELNO1 || MI.TELNO2) as 연락처

FROM PORTAL.MERCHANT_INTRO MI

LEFT JOIN PORTAL.MERCHANT_INTRO_ASP_MAPPING AP ON AP.PLACE_NO = MI.PLACE_NO

INNER JOIN KODASP.BC_MERCHANT BCM ON AP.PLAYER_ID = BCM.MCT_MNG_NO AND SUSPENDED_YN IS NULL

INNER JOIN PORTAL.TBL_SUM_DISCOUNT_APPLICATION DSC ON DSC.PLAYER_ID = AP.PLAYER_NO AND DSC.ASP_NO = 4749

left join portal.agent ag on mi.place_no = ag.player_no

WHERE 1=1

AND MI.LOCALE = 'ko'

AND MI.USE_YN = 'Y'

AND MI.VIEWING_STATUS = 'Y'

AND AP.ASP_NO = 4749

AND MI.MC_TYPE = 'BC'


) A

ORDER BY 지역구, 주소
;
